# Virtex
Download Virtex Via Termux ^_^

CARA INSTALL NYA MELALUI TERMUX


$ pkg update -y

$ pkg upgrade -y

$ pkg install figlet -y 

$ pkg install mc -y

$ pkg install wget -y

$ pkg install git -y

$ pkg install python -y

$ git clone https://github.com/CraZzy-Hacked/Virtex

$ termux-setup-storage

$ cd Virtex

$ pip install requests

$ ls

$ chmod +x *
 
 $ ls
 
 $ python3 Virtex.py
